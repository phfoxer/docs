<Block name="Async/Await">

```swift
let p = Post.keys
do {
    let result = try await Amplify.DataStore.query(Post.self,
                                                   where: p.rating > 4 && p.status == PostStatus.published)
    // result of type [Post]
    print("Published posts with rating greater than 4: \(result)")
} catch {
    print("Error listing posts - \(error)")
}
```

</Block>

<Block name="Combine">

```swift
let p = Post.keys
let sink = Amplify.Publisher.create {
    try await Amplify.DataStore.query(Post.self,
                                      where: p.rating > 4 && p.status == PostStatus.published)
}.sink {
    if case let .failure(error) = $0 {
        print("Error listing posts - \(error)")
    }
}
receiveValue: { result in
    print("Published posts with rating greater than 4: \(result)")
}
```

</Block>

You can also write this in a compositional function manner by replacing the operators with their equivalent predicate statements such as `.gt`, `.and`, etc:

<BlockSwitcher>

<Block name="Async/Await">

```swift
let p = Post.keys
do {
    try await Amplify.DataStore.query(Post.self,
                                      where: p.rating.gt(4).and(p.status.eq(PostStatus.published)))
} catch {
    print("Error listing posts - \(error)")
}
```

</Block>

```swift
let p = Post.keys

    try await Amplify.DataStore.query(
        Post.self,
        where: p.rating > 4 && p.status == PostStatus.published)
}.sink {
    // handle the error like in the previous example
}
receiveValue: {
    //handle the result like in the previous example
}
```
