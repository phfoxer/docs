<Block name="Combine">

```swift
let sink = Amplify.Publisher.create {
    try await Amplify.DataStore.query(Post.self, sort: .ascending(Post.keys.rating))
}.sink {
    if case let .failure(error) = $0 {
        print("Error listing posts - \(error)")
    }
}
receiveValue: { result in
    print("Posts: \(result)")
}
```
