<Block name="Combine">

```swift
let p = Post.keys
let sink = Amplify.Publisher.create {
    try await Amplify.DataStore.query(Post.self, where: p.rating > 4)
}.sink {
    if case let .failure(error) = $0 {
        print("Error listing posts - \(error)")
    }
}
receiveValue: { result in
    print("Posts: \(result)")
}
```
